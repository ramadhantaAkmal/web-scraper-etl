version: '3.7'

x-logging: &default-logging
  driver: json-file
  options:
    max-size: "100m"
    max-file: "5"

services:
  # Airbyte Platform (combines server, webapp, and core services)
  airbyte:
    image: airbyte/airbyte-base-java-image:3  # Use a specific version; check https://hub.docker.com/r/airbyte/airbyte/tags for latest
    container_name: airbyte
    ports:
      - "8000:8000"  # Airbyte UI and API
    env_file:
      - .env
    environment:
      - DATABASE_USER=airbyte
      - DATABASE_PASSWORD=${DB_PASSWORD}
      - DATABASE_HOST=db
      - DATABASE_DB=airbyte
      - DATABASE_URL=jdbc:postgresql://db:5432/airbyte
      - WORKSPACE_ROOT=/data
      - LOCAL_ROOT=/data
      - CONFIG_ROOT=/data
      - WEBAPP_URL=http://localhost:8000
      - API_URL=http://localhost:8000
      # Basic Auth (optional; set to empty strings to disable)
      - BASIC_AUTH_USER=airbyte
      - BASIC_AUTH_PASSWORD=${BASIC_AUTH_PASSWORD}
    volumes:
      - airbyte_data:/data
      - ./workspace:/workspace  # Optional: For custom connectors/secrets
      - ./logs:/logs  # Optional: For logs
    depends_on:
      - db
      - temporal
    restart: unless-stopped
    logging: *default-logging

  # PostgreSQL Database
  db:
    image: postgres:13-alpine
    container_name: airbyte-db
    env_file:
      - .env
    environment:
      - POSTGRES_USER=airbyte
      - POSTGRES_PASSWORD=${DB_PASSWORD}
      - POSTGRES_DB=airbyte
    volumes:
      - db_data:/var/lib/postgresql/data
    restart: unless-stopped
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U airbyte -d airbyte"]
      interval: 5s
      timeout: 5s
      retries: 5
    logging: *default-logging

  # Temporal for Workflow Orchestration
  temporal:
    image: temporalio/auto-setup:1.21.2
    container_name: airbyte-temporal
    env_file:
      - .env
    environment:
      - DB=postgresql
      - DB_PORT=5432
      - POSTGRES_USER=airbyte
      - POSTGRES_PWD=${DB_PASSWORD}
      - POSTGRES_SEEDS=db
    ports:
      - "7233:7233"  # Optional: Temporal UI
    volumes:
      - temporal_data:/etc/temporal
    depends_on:
      db:
        condition: service_healthy
    restart: unless-stopped
    logging: *default-logging

  # Optional: MinIO for Internal Storage (if not using external S3)
  minio:
    image: minio/minio:RELEASE.2025-09-07T16-13-09Z-cpuv1
    container_name: airbyte-minio
    environment:
      - MINIO_ACCESS_KEY=minioadmin
      - MINIO_SECRET_KEY=minioadmin  # Change in production
    ports:
      - "9000:9000"
      - "9001:9001"  # Console
    command: server /data --console-address ":9001"
    volumes:
      - minio_data:/data
    restart: unless-stopped
    logging: *default-logging

volumes:
  airbyte_data:
  db_data:
  temporal_data:
  minio_data:

networks:
  default:
    name: airbyte_network
